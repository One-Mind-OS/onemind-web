## Part 5: The Chat Implementation Deep Dive

### Chat Engine Architecture

```python
# backend/agno/chat/enhanced_chat.py

class EnhancedChatEngine:
    """
    The unified chat interface that combines:
    - Relevance AI: @agent invocation, multi-model, context awareness
    - Motion: Scheduling intelligence, task management
    - Pipedream: Workflow execution from chat
    - OneMind CODEX: Path awareness, intelligence scores
    """

    def __init__(self, user_id: str, session_id: str):
        self.user_id = user_id
        self.session_id = session_id
        self.context = ContextBuilder(user_id)
        self.command_parser = CommandParser()
        self.agent_router = AgentRouter()
        self.workflow_executor = WorkflowExecutor()
        self.citation_tracker = CitationTracker()

    async def process_message(self, message: str) -> ChatResponse:
        """
        Process a user message and return response.

        Message types:
        1. @agent mention → Route to specialist agent
        2. /command → Execute chat command
        3. Multi-step workflow request → Create and execute workflow
        4. Normal chat → Use Legacy with full context
        """

        # 1. Check for @agent mentions
        if self.agent_router.has_agent_mention(message):
            return await self._handle_agent_mention(message)

        # 2. Check for /commands
        if message.startswith('/'):
            return await self._handle_command(message)

        # 3. Check for multi-step workflow patterns
        if self._is_workflow_request(message):
            return await self._handle_workflow_request(message)

        # 4. Normal conversation with full context
        return await self._handle_normal_chat(message)

    async def _handle_agent_mention(self, message: str) -> ChatResponse:
        """Handle @agent mentions like @Researcher, @Analyst"""
        agent_name, task = self.agent_router.parse_mention(message)

        # Get the specialist agent
        agent = await self._get_agent(agent_name)

        # Build context for the agent
        context = await self.context.build_for_agent(agent_name)

        # Execute with the agent
        result = await agent.arun(
            task,
            context=context,
            citation_tracker=self.citation_tracker
        )

        return ChatResponse(
            content=result.content,
            agent=agent_name,
            citations=self.citation_tracker.get_citations(),
            metadata={"agent_used": agent_name}
        )

    async def _handle_command(self, message: str) -> ChatResponse:
        """Handle /commands like /brief, /schedule, /tasks"""
        command, args = self.command_parser.parse(message)

        handlers = {
            'brief': self._cmd_brief,
            'review': self._cmd_review,
            'schedule': self._cmd_schedule,
            'scores': self._cmd_scores,
            'inbox': self._cmd_inbox,
            'tasks': self._cmd_tasks,
            'projects': self._cmd_projects,
            'habits': self._cmd_habits,
            'delegate': self._cmd_delegate,
            'workflow': self._cmd_workflow,
            'remind': self._cmd_remind,
            'quick': self._cmd_quick,
        }

        handler = handlers.get(command)
        if handler:
            return await handler(args)

        return ChatResponse(content=f"Unknown command: /{command}")

    async def _handle_workflow_request(self, message: str) -> ChatResponse:
        """
        Detect and handle multi-step workflow requests.

        Example: "Research competitor XYZ, analyze their pricing,
                 and draft an email to our sales team"
        """
        # Use LLM to parse the request into a workflow plan
        plan = await self._create_workflow_plan(message)

        # Show plan to user for approval
        return ChatResponse(
            content=self._format_workflow_plan(plan),
            action_required="approve_workflow",
            workflow_plan=plan,
            buttons=[
                {"label": "Approve", "action": "approve"},
                {"label": "Modify", "action": "modify"},
                {"label": "Cancel", "action": "cancel"}
            ]
        )

    async def _handle_normal_chat(self, message: str) -> ChatResponse:
        """Normal conversation with Legacy using full context."""

        # Build comprehensive context
        context = await self.context.build_full()

        # Create the prompt with context
        prompt = self._build_contextual_prompt(message, context)

        # Get response from Legacy
        agent = create_legacy_agent(session_id=self.session_id)
        result = await agent.arun(prompt)

        return ChatResponse(
            content=result.content,
            citations=self.citation_tracker.get_citations(),
            context_used=context.summary
        )

    def _build_contextual_prompt(self, message: str, context: UserContext) -> str:
```

```python
class ContextBuilder:
    """
    Builds comprehensive user context for AI interactions.

    This is what makes OneMind different - the AI knows EVERYTHING:
    - All items (tasks, projects, habits, goals)
    - All paths (HP, LE, GE) with current progress
    - Calendar and schedule
    - Intelligence scores
    - Energy levels (from biometrics)
    - Recent conversations
    - Knowledge base
    """

    def __init__(self, user_id: str):
        self.user_id = user_id

    async def build_full(self) -> UserContext:
        """Build complete context for general chat."""

        # Gather all data in parallel
        tasks, scores, calendar, energy, recent = await asyncio.gather(
            self._get_active_tasks(),
            self._get_intelligence_scores(),
            self._get_today_calendar(),
            self._get_energy_level(),
            self._get_recent_context()
        )

        return UserContext(
            user_id=self.user_id,
            user_name=await self._get_user_name(),
            current_date=datetime.now().strftime("%A, %B %d, %Y"),
            current_time=datetime.now().strftime("%I:%M %p"),
            scores=scores,
            today_schedule=self._format_schedule(calendar),
            active_tasks=self._format_tasks(tasks),
            energy_level=energy,
            calendar_availability=self._calculate_availability(calendar),
            recent_context=recent
        )

    async def build_for_agent(self, agent_name: str) -> AgentContext:
        """Build context specific to an agent's needs."""

        # Get agent-specific data
        agent_config = await self._get_agent_config(agent_name)

        # Only include relevant context for this agent
        context_filter = agent_config.context_filter

        tasks = await self._get_filtered_tasks(context_filter)
        knowledge = await self._get_relevant_knowledge(agent_name)

        return AgentContext(
            agent_name=agent_name,
            user_id=self.user_id,
            relevant_tasks=tasks,
            knowledge=knowledge,
            constraints=agent_config.constraints
        )

    async def _get_intelligence_scores(self) -> IntelligenceScores:
        """Get current HPI, LEI, GEI, UI scores."""
        calculator = ScoreCalculator()
        return await calculator.calculate_all(self.user_id)

    async def _get_energy_level(self) -> EnergyLevel:
        """Get current energy level from biometrics or manual input."""
        # Try Oura first
        oura = OuraIntegration()
        try:
            readiness = await oura.get_readiness(self.user_id)
            return EnergyLevel(
                level=self._readiness_to_energy(readiness.score),
                source="oura",
                score=readiness.score
            )
        except:
            # Fall back to manual or default
            return EnergyLevel(level="moderate", source="default")

    async def _get_recent_context(self) -> str:
        """Get recent conversation context and activities."""
        # Last 5 messages
        messages = await self._get_recent_messages(limit=5)

        # Recent task completions
        completions = await self._get_recent_completions(hours=24)

        # Recent captures
        captures = await self._get_recent_captures(limit=3)

        return f"""
Recent conversation:
{self._format_messages(messages)}

Recently completed:
{self._format_completions(completions)}

Recent captures:
{self._format_captures(captures)}
"""
```
